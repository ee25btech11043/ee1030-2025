\begin{enumerate}[label=(\Alph*)]
		\begin{multicols}{4}
		\item $\brak{\frac{1}{3},-\frac{8}{3}}$
		\item $\brak{-\frac{10}{3},-\frac{7}{3}}$
		\item $\brak{-3,-9}$
		\item $\brak{-3,-8}$
		\end{multicols}
\end{enumerate}
